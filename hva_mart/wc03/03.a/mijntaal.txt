<mijntaal>  ::= <commandos>
<commandos>  ::= <commando> | <commando> <nl> <commandos>
<commando>  ::= <led> | <knop>
<led>       ::= "led" <ws> <naam> <ws> <status> <ws>
<knop>      ::=  "knop" <ws> <naam> <ws> <status> <ws>
<label>     ::= "label" <ws> <naam> <ws>
<naam>      ::= <sym> | <sym> <sym>
<sym>       ::= "a" | "b" | ... | "z" | "0" | "1" | ... | "9" | "-" | "_"
<status>    ::= "on" | "off"
<nl>        ::= "" | '000A' <nl> | '000D' <nl>
<ws>        ::= "" | '0020' <ws> | '0009' <ws>
